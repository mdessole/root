if (cuda)
  ROOT_LINKER_LIBRARY(CUDAHist src/RHnCUDA.cu TYPE SHARED DEPENDENCIES MathCore ROOTVecOps)
  target_compile_options(CUDAHist PRIVATE -rdc=true -lineinfo --expt-relaxed-constexpr)
  set_target_properties(CUDAHist PROPERTIES CUDA_SEPARABLE_COMPILATION ON)
  target_compile_definitions(CUDAHist INTERFACE ROOT_RDF_CUDA)
endif()

if (sycl)
  ROOT_LINKER_LIBRARY(SYCLHist 
      src/RHnSYCL.cxx 
      src/RDefH1SYCL.cxx 
    DEPENDENCIES 
      GenVectorSYCL 
      ROOTVecOps
  )
  
  target_compile_definitions(SYCLHist PUBLIC ROOT_RDF_SYCL ROOT_MATH_SYCL)
  if (adaptivecpp)
    target_compile_definitions(SYCLHist PUBLIC ROOT_RDF_ADAPTIVECPP)
  endif()
  add_sycl_library_to_main_root_target(
    TARGET 
      SYCLHist 
    SOURCES 
      src/RHnSYCL.cxx
      src/RDefH1SYCL.cxx 
    COMPILE_DEFINITIONS
      ROOT_RDF_SYCL
      ROOT_MATH_SYCL
  )
endif()
